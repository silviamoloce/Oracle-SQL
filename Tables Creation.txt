
CREATE TABLE Cakes(
    Id NUMBER GENERATED BY DEFAULT AS IDENTITY,
    Name VARCHAR2(50) NOT NULL,
    OwnerBakerId number NOT NULL,
    PRIMARY KEY(Id)
);



CREATE TABLE Bakers(
    Id NUMBER GENERATED BY DEFAULT AS IDENTITY,
    Name VARCHAR2(50) NOT NULL,
    PRIMARY KEY(Id)
);


CREATE TABLE Ingredients(
    Id NUMBER GENERATED BY DEFAULT AS IDENTITY,
    Name VARCHAR2(50) NOT NULL,
    PRIMARY KEY(Id)
);


CREATE TABLE CakeIngredients(
    Id NUMBER GENERATED BY DEFAULT AS IDENTITY,
    CakeId NUMBER NOT NULL,
    IngredientId NUMBER NOT NULL,
    PRIMARY KEY(Id)
);

ALTER TABLE Cakes
ADD CONSTRAINT FK_Bakers
FOREIGN KEY (OwnerBakerId)
REFERENCES Bakers(Id);


ALTER TABLE CakeIngredients
ADD CONSTRAINT FK_CakeIngredients_CakeId
FOREIGN KEY (CakeId)
REFERENCES Cakes(Id);


ALTER TABLE CakeIngredients
ADD CONSTRAINT FK_CakeIngredients_IngredientId
FOREIGN KEY (IngredientId)
REFERENCES Ingredients(Id);
